Opinionated geo utilities for day to day development

## Requirements

- [NodeJS v8.11.1+](https://nodejs.org)
- [npm v5.6.0+](https://www.npmjs.com/)

## Installation

```sh
npm install --save @lykmapipo/geo-tools
```

## Usage

```js
import {
  randomPosition,
  randomPoint,
  readShapefile,
  readGeoJSON
} from '@lykmapipo/geo-tools';

randomPosition()
//=> [-76.41031, 67.0704]

randomPoint()
//=> { type: 'Point', coordinates: [ -76.41031, 67.0704 ] }

readShapefile('data.shp', (error, { finished, feature, next }) => {
  // handle read errors
  if (error) {
    return done(error);
  } 
  // handle read finish
  if (finished) {
    return done();
  } 
  // process features
  // and request next chunk from stream
  processFeature(error => next(error));
});

readGeoJSON('data.geojson', (error, { finished, feature, next }) => {
  // handle read errors
  if (error) {
    return done(error);
  } 
  // handle read finish
  if (finished) {
    return done();
  } 
  // process features
  // and request next chunk from stream
  processFeature(error => next(error));
});
```
